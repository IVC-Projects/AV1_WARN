98 static int read_frame(struct AvxInputContext *input_ctx, aom_image_t *img) {
99  FILE *f = input_ctx->file;
100  y4m_input *y4m = &input_ctx->y4m;
101  int shortread = 0;
102 
103  if (input_ctx->file_type == FILE_TYPE_Y4M) {
104  if (y4m_input_fetch_frame(y4m, f, img) < 1) return 0;
105  } else {
106  shortread = read_yuv_frame(input_ctx, img);
107  }
108 
109  return !shortread;
110 }

984 /* Parses global config arguments into the AvxEncoderConfig. Note that
985  * argv is modified and overwrites all parsed arguments.
986  */
987 static void parse_global_config(struct AvxEncoderConfig *global, int argc,
988  char ***argv) {
989  char **argi, **argj;
990  struct arg arg;
991  const int num_encoder = get_aom_encoder_count();
992  char **argv_local = (char **)*argv;
993 #if CONFIG_FILEOPTIONS
994  int argc_local = argc;
995 #endif
996  if (num_encoder < 1) die("Error: no valid encoder available\n");
997 
998  /* Initialize default parameters */
999  memset(global, 0, sizeof(*global));
1000  global->codec = get_aom_encoder_by_index(num_encoder - 1);
1001  global->passes = 0;
1002  global->color_type = I420;
1003  global->csp = AOM_CSP_UNKNOWN;
1004 
1005 #if CONFIG_FILEOPTIONS
1006  const char *cfg = NULL;
1007  int cfg_included = 0;
1008 #endif
1009  for (argi = argj = argv_local; (*argj = *argi); argi += arg.argv_step) {
1010  arg.argv_step = 1;
1011 
1012 #if CONFIG_FILEOPTIONS
1013  if (arg_match(&arg, &use_cfg, argi)) {
1014  if (cfg_included) continue;
1015  cfg = arg.val;
1016

1740 static void encode_frame(struct stream_state *stream,
1741  struct AvxEncoderConfig *global, struct aom_image *img,
1742  unsigned int frames_in) {
1743  aom_codec_pts_t frame_start, next_frame_start;
1744  struct aom_codec_enc_cfg *cfg = &stream->config.cfg;
1745  struct aom_usec_timer timer;
1746 
1747  frame_start =
1748  (cfg->g_timebase.den * (int64_t)(frames_in - 1) * global->framerate.den) /
1749  cfg->g_timebase.num / global->framerate.num;
1750  next_frame_start =
1751  (cfg->g_timebase.den * (int64_t)(frames_in)*global->framerate.den) /
1752  cfg->g_timebase.num / global->framerate.num;
1753 
1754  /* Scale if necessary */
1755  if (img) {
1756  if ((img->fmt & AOM_IMG_FMT_HIGHBITDEPTH) &&
1757  (img->d_w != cfg->g_w || img->d_h != cfg->g_h)) {
1758  if (img->fmt != AOM_IMG_FMT_I42016) {
1759  fprintf(stderr, "%s can only scale 4:2:0 inputs\n", exec_name);
1760  exit(EXIT_FAILURE);
1761  }
1762 #if CONFIG_LIBYUV
1763  if (!stream->img) {
1764  stream->img =
1765  aom_img_alloc(NULL, AOM_IMG_FMT_I42016, cfg->g_w, cfg->g_h, 16);
1766  }
1767  I420Scale_16(
1768  (uint16_t *)img->planes[AOM_PLANE_Y], img->stride[AOM_PLANE_Y] / 2,
1769  (uint16_t *)img->planes[AOM_PLANE_U], img->stride[AOM_PLANE_U] / 2,
1770  (uint16_t *)img->planes[AOM_PLANE_V], img->stride[AOM_PLANE_V] / 2,
1771  img->d_w, img->d_h, (uint16_t *)stream->img->planes[AOM_PLANE_Y],
1772  stream->img->stride[AOM_PLANE_Y] / 2,
1773  (uint16_t *)stream->img->planes[AOM_PLANE_U],
1774  stream->img->stride[AOM_PLANE_U] / 2,
1775  (uint16_t *)stream->img->planes[AOM_PLANE_V],
1776  stream->img->stride[AOM_PLANE_V] / 2, stream->img->d_w,
1777  stream->img->d_h, kFilterBox);
1778  img = stream->img;
1779 #else
1780  stream->encoder.err = 1;
1781  ctx_exit_on_error(&stream->encoder,
1782  "Stream %d: Failed to encode frame.\n"
1783  "libyuv is required for scaling but is currently "
1784  "disabled.\n"
1785  "Be sure to specify -DCONFIG_LIBYUV=1 when running "
1786  "cmake.\n",
1787  stream->index);
1788 #endif
1789  }
1790  }
1791  if (img && (img->d_w != cfg->g_w || img->d_h != cfg->g_h)) {
1792  if (img->fmt != AOM_IMG_FMT_I420 && img->fmt != AOM_IMG_FMT_YV12) {
1793  fprintf(stderr, "%s can only scale 4:2:0 8bpp inputs\n", exec_name);
1794  exit(EXIT_FAILURE);
1795  }
1796 #if CONFIG_LIBYUV
1797  if (!stream->img)
1798  stream->img =
1799  aom_img_alloc(NULL, AOM_IMG_FMT_I420, cfg->g_w, cfg->g_h, 16);
1800  I420Scale(
1801  img->planes[AOM_PLANE_Y], img->stride[AOM_PLANE_Y],
1802  img->planes[AOM_PLANE_U], img->stride[AOM_PLANE_U],
1803  img->planes[AOM_PLANE_V], img->stride[AOM_PLANE_V], img->d_w, img->d_h,
1804  stream->img->planes[AOM_PLANE_Y], stream->img->stride[AOM_PLANE_Y],
1805  stream->img->planes[AOM_PLANE_U], stream->img->stride[AOM_PLANE_U],
1806  stream->img->planes[AOM_PLANE_V], stream->img->stride[AOM_PLANE_V],
1807  stream->img->d_w, stream->img->d_h, kFilterBox);
1808  img = stream->img;
1809 #else
1810  stream->encoder.err = 1;
1811  ctx_exit_on_error(&stream->encoder,
1812  "Stream %d: Failed to encode frame.\n"
1813  "Scaling disabled in this configuration. \n"
1814  "To enable, configure with --enable-libyuv\n",
1815  stream->index);
1816 #endif
1817  }
1818 
1819  aom_usec_timer_start(&timer);
1820  aom_codec_encode(&stream->encoder, img, frame_start,
1821  (uint32_t)(next_frame_start - frame_start), 0);
1822  aom_usec_timer_mark(&timer);
1823  stream->cx_time += aom_usec_timer_elapsed(&timer);
1824  ctx_exit_on_error(&stream->encoder, "Stream %d: Failed to encode frame",
1825  stream->index);
1826 }
