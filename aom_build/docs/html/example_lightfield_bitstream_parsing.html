12 // Lightfield Bitstream Parsing
13 // ============================
14 //
15 // This is a lightfield bitstream parsing example. It takes an input file
16 // containing the whole compressed lightfield bitstream(ivf file) and a text
17 // file containing a stream of tiles to decode and then constructs and outputs
18 // a new bitstream that can be decoded by an AV1 decoder. The output bitstream
19 // contains reference frames(i.e. anchor frames), camera frame header, and
20 // tile list OBUs. num_references is the number of anchor frames coded at the
21 // beginning of the light field file. After running the lightfield encoder,
22 // run lightfield bitstream parsing:
23 // examples/lightfield_bitstream_parsing vase10x10.ivf vase_tile_list.ivf 4
24 // tile_list.txt
25 //
26 // The tile_list.txt is expected to be of the form:
27 // Frame <frame_index0>
28 // <image_index0> <anchor_index0> <tile_col0> <tile_row0>
29 // <image_index1> <anchor_index1> <tile_col1> <tile_row1>
30 // ...
31 // Frame <frame_index1)
32 // ...
33 //
34 // The "Frame" markers indicate a new render frame and thus a new tile list
35 // will be started and the old one flushed. The image_indexN, anchor_indexN,
36 // tile_colN, and tile_rowN identify an individual tile to be decoded and
37 // to use anchor_indexN anchor image for MCP.

85 static int get_image_bps(aom_img_fmt_t fmt) {
86  switch (fmt) {
87  case AOM_IMG_FMT_I420: return 12;
88  case AOM_IMG_FMT_I422: return 16;
89  case AOM_IMG_FMT_I444: return 24;
90  case AOM_IMG_FMT_I42016: return 24;
91  case AOM_IMG_FMT_I42216: return 32;
92  case AOM_IMG_FMT_I44416: return 48;
93  default: die("Invalid image format");
94  }
95  return 0;
96 }

98 void process_tile_list(const TILE_LIST_INFO *tiles, int num_tiles,
99  aom_codec_pts_t tl_pts, unsigned char **frames,
100  const size_t *frame_sizes, aom_codec_ctx_t *codec,
101  unsigned char *tl_buf, AvxVideoWriter *writer,
102  uint8_t output_frame_width_in_tiles_minus_1,
103  uint8_t output_frame_height_in_tiles_minus_1) {
104  unsigned char *tl = tl_buf;
105  struct aom_write_bit_buffer wb = { tl, 0 };
106  unsigned char *saved_obu_size_loc = NULL;
107  uint32_t tile_list_obu_header_size = 0;
108  uint32_t tile_list_obu_size = 0;
109  int num_tiles_minus_1 = num_tiles - 1;
110  int i;
111 
112  // Write the tile list OBU header that is 1 byte long.
113  aom_wb_write_literal(&wb, 0, 1); // forbidden bit.
114  aom_wb_write_literal(&wb, 8, 4); // tile list OBU: "1000"
115  aom_wb_write_literal(&wb, 0, 1); // obu_extension = 0
116  aom_wb_write_literal(&wb, 1, 1); // obu_has_size_field
117  aom_wb_write_literal(&wb, 0, 1); // reserved
118  tl++;
119  tile_list_obu_header_size++;
120 
121  // Write the OBU size using a fixed length_field_size of 4 bytes.
122  saved_obu_size_loc = tl;
123  // aom_wb_write_unsigned_literal(&wb, data, bits) requires that bits <= 32.
124  aom_wb_write_unsigned_literal(&wb, 0, 32);
125  tl += 4;
126  tile_list_obu_header_size += 4;
127 
128  // write_tile_list_obu()
129  aom_wb_write_literal(&wb, output_frame_width_in_tiles_minus_1, 8);
130  aom_wb_write_literal(&wb, output_frame_height_in_tiles_minus_1, 8);
131  aom_wb_write_literal(&wb, num_tiles_minus_1, 16);
132  tl += 4;
133  tile_list_obu_size += 4;
134 
135  // Write each tile's data
136  for (i = 0; i <= num_tiles_minus_1; i++) {
137  aom_tile_data tile_data = { 0, NULL, 0 };
138 
139  int image_idx = tiles[i].image_idx;
140  int ref_idx = tiles[i].reference_idx;
141  int tc = tiles[i].tile_col;
142  int tr = tiles[i].tile_row;
143 
144  // Reset bit writer to the right location.
145  wb.bit_buffer = tl;
146  wb.bit_offset = 0;
147 
148  size_t frame_size = frame_sizes[image_idx];
149  const unsigned char *frame = frames[image_idx];
150 
151  aom_codec_control_(codec, AV1_SET_DECODE_TILE_ROW, tr);
152  aom_codec_control_(codec, AV1_SET_DECODE_TILE_COL, tc);
153 
154  aom_codec_err_t aom_status =
155  aom_codec_decode(codec, frame, frame_size, NULL);
156  if (aom_status) die_codec(codec, "Failed to decode tile.");
157 
158  aom_codec_control_(codec, AV1D_GET_TILE_DATA, &tile_data);
159 
160  // Copy over tile info.
161  // uint8_t anchor_frame_idx;
162  // uint8_t tile_row;
163  // uint8_t tile_col;
164  // uint16_t coded_tile_data_size_minus_1;
165  // uint8_t *coded_tile_data;
166  uint32_t tile_info_bytes = 5;
167  aom_wb_write_literal(&wb, ref_idx, 8);
168  aom_wb_write_literal(&wb, tr, 8);
169  aom_wb_write_literal(&wb, tc, 8);
170  aom_wb_write_literal(&wb, (int)tile_data.coded_tile_data_size - 1, 16);
171  tl += tile_info_bytes;
172 
173  memcpy(tl, (uint8_t *)tile_data.coded_tile_data,
174  tile_data.coded_tile_data_size);
175  tl += tile_data.coded_tile_data_size;
176 
177  tile_list_obu_size +=
178  tile_info_bytes + (uint32_t)tile_data.coded_tile_data_size;
179  }
180 
181  // Write tile list OBU size.
182  size_t bytes_written = 0;
183  if (aom_uleb_encode_fixed_size(tile_list_obu_size, 4, 4, saved_obu_size_loc,
184  &bytes_written))
185  die_codec(codec, "Failed to encode the tile list obu size.");
186 
187  // Copy the tile list.
188  if (!aom_video_writer_write_frame(
189  writer, tl_buf, tile_list_obu_header_size + tile_list_obu_size,
190  tl_pts))
191  die_codec(codec, "Failed to copy compressed tile list.");
192 }

194 int main(int argc, char **argv) {
195  aom_codec_ctx_t codec;
196  AvxVideoReader *reader = NULL;
197  AvxVideoWriter *writer = NULL;
198  const AvxInterface *decoder = NULL;
199  const AvxVideoInfo *info = NULL;
200  int num_references;
201  int i;
202  aom_codec_pts_t pts;
203  const char *tile_list_file = NULL;
204 
205  exec_name = argv[0];
206  if (argc != 5) die("Invalid number of arguments.");
207 
208  reader = aom_video_reader_open(argv[1]);
209  if (!reader) die("Failed to open %s for reading.", argv[1]);
210 
211  num_references = (int)strtol(argv[3], NULL, 0);
212  info = aom_video_reader_get_info(reader);
213 
214  // The writer to write out ivf file in tile list OBU, which can be decoded by
215  // AV1 decoder.
216  writer = aom_video_writer_open(argv[2], kContainerIVF, info);
217  if (!writer) die("Failed to open %s for writing", argv[2]);
218 
219  tile_list_file = argv[4];
220 
221  decoder = get_aom_decoder_by_fourcc(info->codec_fourcc);
222  if (!decoder) die("Unknown input codec.");
223  printf("Using %s\n", aom_codec_iface_name(decoder->codec_interface()));
224 
225  if (aom_codec_dec_init(&codec, decoder->codec_interface(), NULL, 0))
226  die_codec(&codec, "Failed to initialize decoder.");
227 
228  // Decode anchor frames.
229  aom_codec_control_(&codec, AV1_SET_TILE_MODE, 0);
230 
231  printf("Reading %d reference images.\n", num_references);
232  for (i = 0; i < num_references; ++i) {
233  aom_video_reader_read_frame(reader);
234 
235  size_t frame_size = 0;
236  const unsigned char *frame =
237  aom_video_reader_get_frame(reader, &frame_size);
238  pts = (aom_codec_pts_t)aom_video_reader_get_frame_pts(reader);
239 
240  // Copy references bitstream directly.
241  if (!aom_video_writer_write_frame(writer, frame, frame_size, pts))
242  die_codec(&codec, "Failed to copy compressed anchor frame.");
243 
244  if (aom_codec_decode(&codec, frame, frame_size, NULL))
245  die_codec(&codec, "Failed to decode frame.");
246  }
247 
248  // Decode camera frames.
249  aom_codec_control_(&codec, AV1_SET_TILE_MODE, 1);
250  aom_codec_control_(&codec, AV1D_EXT_TILE_DEBUG, 1);
251 
252  FILE *infile = aom_video_reader_get_file(reader);
253  // Record the offset of the first camera image.
254  const FileOffset camera_frame_pos = ftello(infile);
255 
256  printf("Loading compressed frames into memory.\n");
257 
258  // Count the frames in the lightfield.
259  int num_frames = 0;
260  while (aom_video_reader_read_frame(reader)) {
261  ++num_frames;
262  }
263  if (num_frames < 1) die("Input light field has no frames.");
264 
265  // Read all of the lightfield frames into memory.
266  unsigned char **frames =
267  (unsigned char **)malloc(num_frames * sizeof(unsigned char *));
268  size_t *frame_sizes = (size_t *)malloc(num_frames * sizeof(size_t));
269  // Seek to the first camera image.
270  fseeko(infile, camera_frame_pos, SEEK_SET);
271  for (int f = 0; f < num_frames; ++f) {
272  aom_video_reader_read_frame(reader);
273  size_t frame_size = 0;
274  const unsigned char *frame =
275  aom_video_reader_get_frame(reader, &frame_size);
276  frames[f] = (unsigned char *)malloc(frame_size * sizeof(unsigned char));
277  memcpy(frames[f], frame, frame_size);
278  frame_sizes[f] = frame_size;
279  }
280  printf("Read %d frames.\n", num_frames);
281 
282  // Copy first camera frame for getting camera frame header. This is done
283  // only once.
284  {
285  size_t frame_size = frame_sizes[0];
286  const unsigned char *frame = frames[0];
287  pts = num_references;
288  aom_tile_data frame_header_info = { 0, NULL, 0 };
289 
290  // Need to decode frame header to get camera frame header info. So, here
291  // decoding 1 tile is enough.
292  aom_codec_control_(&codec, AV1_SET_DECODE_TILE_ROW, 0);
293  aom_codec_control_(&codec, AV1_SET_DECODE_TILE_COL, 0);
294 
295  aom_codec_err_t aom_status =
296  aom_codec_decode(&codec, frame, frame_size, NULL);
297  if (aom_status) die_codec(&codec, "Failed to decode tile.");
298 
299  aom_codec_control_(&codec, AV1D_GET_FRAME_HEADER_INFO, &frame_header_info);
300 
301  size_t obu_size_offset =
302  (uint8_t *)frame_header_info.coded_tile_data - frame;
303  size_t length_field_size = frame_header_info.coded_tile_data_size;
304  // Remove ext-tile tile info.
305  uint32_t frame_header_size = (uint32_t)frame_header_info.extra_size - 1;
306  size_t bytes_to_copy =
307  obu_size_offset + length_field_size + frame_header_size;
308 
309  unsigned char *frame_hdr_buf = (unsigned char *)malloc(bytes_to_copy);
310  if (frame_hdr_buf == NULL)
311  die_codec(&codec, "Failed to allocate frame header buffer.");
312 
313  memcpy(frame_hdr_buf, frame, bytes_to_copy);
314 
315  // Update frame header OBU size.
316  size_t bytes_written = 0;
317  if (aom_uleb_encode_fixed_size(
318  frame_header_size, length_field_size, length_field_size,
319  frame_hdr_buf + obu_size_offset, &bytes_written))
320  die_codec(&codec, "Failed to encode the tile list obu size.");
321 
322  // Copy camera frame header bitstream.
323  if (!aom_video_writer_write_frame(writer, frame_hdr_buf, bytes_to_copy,
324  pts))
325  die_codec(&codec, "Failed to copy compressed camera frame header.");
326  free(frame_hdr_buf);
327  }
328 
329  // Read out the image format.
330  aom_img_fmt_t ref_fmt = 0;
331  if (aom_codec_control(&codec, AV1D_GET_IMG_FORMAT, &ref_fmt))
332  die_codec(&codec, "Failed to get the image format");
333  const int bps = get_image_bps(ref_fmt);
334  if (!bps) die_codec(&codec, "Invalid image format.");
335  // read out the tile size.
336  unsigned int tile_size = 0;
337  if (aom_codec_control(&codec, AV1D_GET_TILE_SIZE, &tile_size))
338  die_codec(&codec, "Failed to get the tile size");
339  const unsigned int tile_width = tile_size >> 16;
340  const unsigned int tile_height = tile_size & 65535;
341  // Allocate a buffer to store tile list bitstream.
342  const size_t data_sz = MAX_TILES * ALIGN_POWER_OF_TWO(tile_width, 5) *
343  ALIGN_POWER_OF_TWO(tile_height, 5) * bps / 8;
344 
345  unsigned char *tl_buf = (unsigned char *)malloc(data_sz);
346  if (tl_buf == NULL) die_codec(&codec, "Failed to allocate tile list buffer.");
347 
348  aom_codec_pts_t tl_pts = num_references;
349  const uint8_t output_frame_width_in_tiles_minus_1 =
350  output_frame_width / tile_width - 1;
351  const uint8_t output_frame_height_in_tiles_minus_1 =
352  output_frame_height / tile_height - 1;
353 
354  printf("Reading tile list from file.\n");
355  char line[1024];
356  FILE *tile_list_fptr = fopen(tile_list_file, "r");
357  if (!tile_list_fptr) die_codec(&codec, "Failed to open tile list file.");
358  int num_tiles = 0;
359  TILE_LIST_INFO tiles[MAX_TILES];
360  while ((fgets(line, 1024, tile_list_fptr)) != NULL) {
361  if (line[0] == 'F' || num_tiles >= MAX_TILES) {
362  // Flush existing tile list and start another, either because we hit a
363  // new render frame or because we've hit our max number of tiles per list.
364  if (num_tiles > 0) {
365  process_tile_list(tiles, num_tiles, tl_pts, frames, frame_sizes, &codec,
366  tl_buf, writer, output_frame_width_in_tiles_minus_1,
367  output_frame_height_in_tiles_minus_1);
368  ++tl_pts;
369  }
370  num_tiles = 0;
371  }
372  if (line[0] == 'F') {
373  continue;
374  }
375  if (sscanf(line, "%d %d %d %d", &tiles[num_tiles].image_idx,
376  &tiles[num_tiles].reference_idx, &tiles[num_tiles].tile_col,
377  &tiles[num_tiles].tile_row) == 4) {
378  if (tiles[num_tiles].image_idx >= num_frames) {
379  die("Tile list image_idx out of bounds: %d >= %d.",
380  tiles[num_tiles].image_idx, num_frames);
381  }
382  if (tiles[num_tiles].reference_idx >= num_references) {
383  die("Tile list reference_idx out of bounds: %d >= %d.",
384  tiles[num_tiles].reference_idx, num_references);
385  }
386  ++num_tiles;
387  }
388  }
389  if (num_tiles > 0) {
390  // Flush out the last tile list.
391  process_tile_list(tiles, num_tiles, tl_pts, frames, frame_sizes, &codec,
392  tl_buf, writer, output_frame_width_in_tiles_minus_1,
393  output_frame_height_in_tiles_minus_1);
394  ++tl_pts;
395  }
396 
397  const int num_tile_lists = (int)(tl_pts - pts);
398  printf("Finished processing tile lists. Num tile lists: %d.\n",
399  num_tile_lists);
400  free(tl_buf);
401  for (int f = 0; f < num_frames; ++f) {
402  free(frames[f]);
403  }
404  free(frame_sizes);
405  free(frames);
406  if (aom_codec_destroy(&codec)) die_codec(&codec, "Failed to destroy codec");
407  aom_video_writer_close(writer);
408  aom_video_reader_close(reader);
409 
410  return EXIT_SUCCESS;
411 }
