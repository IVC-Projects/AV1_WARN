12 // Lightfield Decoder
13 // ==================
14 //
15 // This is an example of a simple lightfield decoder. It builds upon the
16 // simple_decoder.c example. It takes an input file containing the compressed
17 // data (in ivf format), treating it as a lightfield instead of a video; and a
18 // text file with a list of tiles to decode. There is an optional parameter
19 // allowing to choose the output format, and the supported formats are
20 // YUV1D(default), YUV, and NV12.
21 // After running the lightfield encoder, run lightfield decoder to decode a
22 // batch of tiles:
23 // examples/lightfield_decoder vase10x10.ivf vase_reference.yuv 4 tile_list.txt
24 // 0(optional)
25 // The tile_list.txt is expected to be of the form:
26 // Frame <frame_index0>
27 // <image_index0> <anchor_index0> <tile_col0> <tile_row0>
28 // <image_index1> <anchor_index1> <tile_col1> <tile_row1>
29 // ...
30 // Frame <frame_index1)
31 // ...
32 //
33 // The "Frame" markers indicate a new render frame and thus a new tile list
34 // will be started and the old one flushed. The image_indexN, anchor_indexN,
35 // tile_colN, and tile_rowN identify an individual tile to be decoded and
36 // to use anchor_indexN anchor image for MCP.

63 static void aom_img_copy_tile(const aom_image_t *src, const aom_image_t *dst,
64  int dst_row_offset, int dst_col_offset) {
65  const int shift = (src->fmt & AOM_IMG_FMT_HIGHBITDEPTH) ? 1 : 0;
66  int plane;
67 
68  for (plane = 0; plane < 3; ++plane) {
69  const unsigned char *src_buf = src->planes[plane];
70  const int src_stride = src->stride[plane];
71  unsigned char *dst_buf = dst->planes[plane];
72  const int dst_stride = dst->stride[plane];
73  const int roffset =
74  (plane > 0) ? dst_row_offset >> dst->y_chroma_shift : dst_row_offset;
75  const int coffset =
76  (plane > 0) ? dst_col_offset >> dst->x_chroma_shift : dst_col_offset;
77 
78  // col offset needs to be adjusted for HBD.
79  dst_buf += roffset * dst_stride + (coffset << shift);
80 
81  const int w = (aom_img_plane_width(src, plane) << shift);
82  const int h = aom_img_plane_height(src, plane);
83  int y;
84 
85  for (y = 0; y < h; ++y) {
86  memcpy(dst_buf, src_buf, w);
87  src_buf += src_stride;
88  dst_buf += dst_stride;
89  }
90  }
91 }

93 void decode_tile(aom_codec_ctx_t *codec, const unsigned char *frame,
94  size_t frame_size, int tr, int tc, int ref_idx,
95  aom_image_t *reference_images, aom_image_t *output,
96  int *tile_idx, unsigned int *output_bit_depth,
97  aom_image_t **img_ptr, int output_format) {
98  aom_codec_control_(codec, AV1_SET_TILE_MODE, 1);
99  aom_codec_control_(codec, AV1D_EXT_TILE_DEBUG, 1);
100  aom_codec_control_(codec, AV1_SET_DECODE_TILE_ROW, tr);
101  aom_codec_control_(codec, AV1_SET_DECODE_TILE_COL, tc);
102 
103  av1_ref_frame_t ref;
104  ref.idx = 0;
105  ref.use_external_ref = 1;
106  ref.img = reference_images[ref_idx];
107  if (aom_codec_control(codec, AV1_SET_REFERENCE, &ref)) {
108  die_codec(codec, "Failed to set reference frame.");
109  }
110 
111  aom_codec_err_t aom_status = aom_codec_decode(codec, frame, frame_size, NULL);
112  if (aom_status) die_codec(codec, "Failed to decode tile.");
113 
114  aom_codec_iter_t iter = NULL;
115  aom_image_t *img = aom_codec_get_frame(codec, &iter);
116  if (!img) die_codec(codec, "Failed to get frame.");
117  *img_ptr = img;
118 
119  // aom_img_alloc() sets bit_depth as follows:
120  // output->bit_depth = (fmt & AOM_IMG_FMT_HIGHBITDEPTH) ? 16 : 8;
121  // Use img->bit_depth(read from bitstream), so that aom_shift_img()
122  // works as expected.
123  output->bit_depth = img->bit_depth;
124  *output_bit_depth = img->bit_depth;
125 
126  if (output_format != YUV1D) {
127  // read out the tile size.
128  unsigned int tile_size = 0;
129  if (aom_codec_control(codec, AV1D_GET_TILE_SIZE, &tile_size))
130  die_codec(codec, "Failed to get the tile size");
131  const unsigned int tile_width = tile_size >> 16;
132  const unsigned int tile_height = tile_size & 65535;
133  const uint8_t output_frame_width_in_tiles = output_frame_width / tile_width;
134 
135  // Copy the tile to the output frame.
136  const int row_offset =
137  (*tile_idx / output_frame_width_in_tiles) * tile_height;
138  const int col_offset =
139  (*tile_idx % output_frame_width_in_tiles) * tile_width;
140 
141  aom_img_copy_tile(img, output, row_offset, col_offset);
142  (*tile_idx)++;
143  }
144 }

156 int main(int argc, char **argv) {
157  FILE *outfile = NULL;
158  aom_codec_ctx_t codec;
159  AvxVideoReader *reader = NULL;
160  const AvxInterface *decoder = NULL;
161  const AvxVideoInfo *info = NULL;
162  int num_references;
163  aom_img_fmt_t ref_fmt = 0;
164  aom_image_t reference_images[MAX_EXTERNAL_REFERENCES];
165  aom_image_t output;
166  aom_image_t *output_shifted = NULL;
167  size_t frame_size = 0;
168  const unsigned char *frame = NULL;
169  int i, j;
170  const char *tile_list_file = NULL;
171  int output_format = YUV1D;
172  exec_name = argv[0];
173 
174  if (argc < 5) die("Invalid number of arguments.");
175 
176  reader = aom_video_reader_open(argv[1]);
177  if (!reader) die("Failed to open %s for reading.", argv[1]);
178 
179  if (!(outfile = fopen(argv[2], "wb")))
180  die("Failed to open %s for writing.", argv[2]);
181 
182  num_references = (int)strtol(argv[3], NULL, 0);
183  tile_list_file = argv[4];
184 
185  if (argc > 5) output_format = (int)strtol(argv[5], NULL, 0);
186  if (output_format < YUV1D || output_format > NV12)
187  die("Output format out of range [0, 2]");
188 
189  info = aom_video_reader_get_info(reader);
190 
191  decoder = get_aom_decoder_by_fourcc(info->codec_fourcc);
192  if (!decoder) die("Unknown input codec.");
193  printf("Using %s\n", aom_codec_iface_name(decoder->codec_interface()));
194 
195  if (aom_codec_dec_init(&codec, decoder->codec_interface(), NULL, 0))
196  die_codec(&codec, "Failed to initialize decoder.");
197 
198  if (aom_codec_control(&codec, AV1D_SET_IS_ANNEXB, info->is_annexb)) {
199  die("Failed to set annex b status");
200  }
201 
202  // Decode anchor frames.
203  aom_codec_control_(&codec, AV1_SET_TILE_MODE, 0);
204  for (i = 0; i < num_references; ++i) {
205  aom_video_reader_read_frame(reader);
206  frame = aom_video_reader_get_frame(reader, &frame_size);
207  if (aom_codec_decode(&codec, frame, frame_size, NULL))
208  die_codec(&codec, "Failed to decode frame.");
209 
210  if (i == 0) {
211  if (aom_codec_control(&codec, AV1D_GET_IMG_FORMAT, &ref_fmt))
212  die_codec(&codec, "Failed to get the image format");
213 
214  int frame_res[2];
215  if (aom_codec_control(&codec, AV1D_GET_FRAME_SIZE, frame_res))
216  die_codec(&codec, "Failed to get the image frame size");
217 
218  // Allocate memory to store decoded references. Allocate memory with the
219  // border so that it can be used as a reference.
220  for (j = 0; j < num_references; j++) {
221  unsigned int border = AOM_BORDER_IN_PIXELS;
222  if (!aom_img_alloc_with_border(&reference_images[j], ref_fmt,
223  frame_res[0], frame_res[1], 32, 8,
224  border)) {
225  die("Failed to allocate references.");
226  }
227  }
228  }
229 
230  if (aom_codec_control(&codec, AV1_COPY_NEW_FRAME_IMAGE,
231  &reference_images[i]))
232  die_codec(&codec, "Failed to copy decoded reference frame");
233 
234  aom_codec_iter_t iter = NULL;
235  aom_image_t *img = NULL;
236  while ((img = aom_codec_get_frame(&codec, &iter)) != NULL) {
237  char name[1024];
238  snprintf(name, sizeof(name), "ref_%d.yuv", i);
239  printf("writing ref image to %s, %d, %d\n", name, img->d_w, img->d_h);
240  FILE *ref_file = fopen(name, "wb");
241  aom_img_write(img, ref_file);
242  fclose(ref_file);
243  }
244  }
245 
246  FILE *infile = aom_video_reader_get_file(reader);
247  // Record the offset of the first camera image.
248  const FileOffset camera_frame_pos = ftello(infile);
249 
250  printf("Loading compressed frames into memory.\n");
251 
252  // Count the frames in the lightfield.
253  int num_frames = 0;
254  while (aom_video_reader_read_frame(reader)) {
255  ++num_frames;
256  }
257  if (num_frames < 1) die("Input light field has no frames.");
258 
259  // Read all of the lightfield frames into memory.
260  unsigned char **frames =
261  (unsigned char **)malloc(num_frames * sizeof(unsigned char *));
262  size_t *frame_sizes = (size_t *)malloc(num_frames * sizeof(size_t));
263  // Seek to the first camera image.
264  fseeko(infile, camera_frame_pos, SEEK_SET);
265  for (int f = 0; f < num_frames; ++f) {
266  aom_video_reader_read_frame(reader);
267  frame = aom_video_reader_get_frame(reader, &frame_size);
268  frames[f] = (unsigned char *)malloc(frame_size * sizeof(unsigned char));
269  memcpy(frames[f], frame, frame_size);
270  frame_sizes[f] = frame_size;
271  }
272  printf("Read %d frames.\n", num_frames);
273 
274  if (output_format != YUV1D) {
275  // Allocate the output frame.
276  aom_img_fmt_t out_fmt = ref_fmt;
277  if (!CONFIG_LOWBITDEPTH) out_fmt |= AOM_IMG_FMT_HIGHBITDEPTH;
278  if (!aom_img_alloc(&output, out_fmt, output_frame_width,
279  output_frame_height, 32))
280  die("Failed to allocate output image.");
281  }
282 
283  printf("Decoding tile list from file.\n");
284  char line[1024];
285  FILE *tile_list_fptr = fopen(tile_list_file, "r");
286  int tile_list_cnt = 0;
287  int tile_list_writes = 0;
288  int tile_idx = 0;
289  aom_image_t *out = NULL;
290  unsigned int output_bit_depth = 0;
291 
292  while ((fgets(line, 1024, tile_list_fptr)) != NULL) {
293  if (line[0] == 'F') {
294  if (output_format != YUV1D) {
295  // Write out the tile list.
296  if (tile_list_cnt) {
297  out = &output;
298  if (output_bit_depth != 0)
299  aom_shift_img(output_bit_depth, &out, &output_shifted);
300  img_write_to_file(out, outfile, output_format);
301  tile_list_writes++;
302  }
303 
304  tile_list_cnt++;
305  tile_idx = 0;
306  // Then memset the frame.
307  memset(output.img_data, 0, output.sz);
308  }
309  continue;
310  }
311 
312  int image_idx, ref_idx, tc, tr;
313  sscanf(line, "%d %d %d %d", &image_idx, &ref_idx, &tc, &tr);
314  if (image_idx >= num_frames) {
315  die("Tile list image_idx out of bounds: %d >= %d.", image_idx,
316  num_frames);
317  }
318  if (ref_idx >= num_references) {
319  die("Tile list ref_idx out of bounds: %d >= %d.", ref_idx,
320  num_references);
321  }
322  frame = frames[image_idx];
323  frame_size = frame_sizes[image_idx];
324 
325  aom_image_t *img = NULL;
326  decode_tile(&codec, frame, frame_size, tr, tc, ref_idx, reference_images,
327  &output, &tile_idx, &output_bit_depth, &img, output_format);
328  if (output_format == YUV1D) {
329  out = img;
330  if (output_bit_depth != 0)
331  aom_shift_img(output_bit_depth, &out, &output_shifted);
332  aom_img_write(out, outfile);
333  }
334  }
335 
336  if (output_format != YUV1D) {
337  // Write out the last tile list.
338  if (tile_list_writes < tile_list_cnt) {
339  out = &output;
340  if (output_bit_depth != 0)
341  aom_shift_img(output_bit_depth, &out, &output_shifted);
342  img_write_to_file(out, outfile, output_format);
343  }
344  }
345 
346  if (output_shifted) aom_img_free(output_shifted);
347  if (output_format != YUV1D) aom_img_free(&output);
348  for (i = 0; i < num_references; i++) aom_img_free(&reference_images[i]);
349  for (int f = 0; f < num_frames; ++f) {
350  free(frames[f]);
351  }
352  free(frame_sizes);
353  free(frames);
354  if (aom_codec_destroy(&codec)) die_codec(&codec, "Failed to destroy codec");
355  aom_video_reader_close(reader);
356  fclose(outfile);
357 
358  return EXIT_SUCCESS;
359 }
